Dict -Słownik, nawiasy klamrowe {}

Struktura zbudowana na zasadzie klucz (key): wartość (value).

W słowniku nie możemy mieć dwóch identycznych kluczy

państwa_miasta = {"Polska": "Warszawa",

"Austria": "Wiedeń",

"Niemcy": "Berlin",

"Australia":"Sydney"}

#dodawanie

państwa_miasta ['Czechy']= 'Praga'

#drukowanie kluczy – iterowanie po kluczu

for panstwo in państwa_miasta.keys():

print(panstwo)

#drukowanie wartości - iterowanie po wartości

for stolica in państwa_miasta.values():

print(stolica)

#drukowanie klucza i wartości przez items

for panstwo,stolica in państwa_miasta.items():

print(panstwo + "-" + stolica)

#szukanie wartości dla klucza

print(państwa_miasta['Polska'])

#output Warszawa

państw_miasta = {"Polska": "Warszawa",

"Austria": "Wiedeń",

"Niemcy": "Berlin",

"Australia":"Sydney"}

# set default - dodaje zbiór do elementu, setdefault wstawia klucz z podaną wortąscią , również podmienia stary klucz z nową wartością

print(państw_miasta.setdefault("USA","Washington DC"))

# usuwanie zbioru metodą pop

państw_miasta.pop("USA","Washington DC")

#pop items - usuwa ostani zestaw wartości dodanych do słownika

print(panstw_miasta.popitem())

#szukanie klucza w słowniku

if "Polska" in panstw_miasta.keys():

print ("znaleziono")

else:

print ("Nie ma podanego klucza w słowniku")

#czyszczenie słownika

panstw_miasta.clear()

#what is output of the code

fib = {1: 1, 2: 1, 3: 2, 4: 3}

print(fib.get(4, 0) + fib.get(7, 5))

# dictionary.get(key, optional_value) So... fib.get(4,0) = 3 din.get(7,5) = 5

# key value 7 not exists, so optional value 5 taken. Adding both will give the answer.

#--> 3 + 5 = 8
